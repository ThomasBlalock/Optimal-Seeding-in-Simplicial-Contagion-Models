\begin{frame}{Motivation: From Theory to Application}
    \begin{itemize}
        \item \textbf{Real-World Application:} Developing a smart notification algorithm for the social app Sidequest to maximize new event attendance.
        \item \textbf{The Core Problem:} How do we optimally select initial "seed" users to trigger a widespread cascade of engagement?
        \item \textbf{The Value of this Replication:}
        \begin{itemize}
            \item Simple contagion wrongly assumes that group interactions behave equivalently to combined individual interactions.
            \item Iacopini et al. provides the foundational mathematical framework for modeling these group-level influence thresholds.
        \end{itemize}
        \item \textbf{Ultimate Goal:} Understand simplicial contagion and optimize virality over it.
    \end{itemize}
\end{frame}

% Slide 2
\begin{frame}{The Simplicial Contagion Model (SCM)}
    \begin{itemize}
        \item \textbf{Simplex Definition:} A $k$-simplex $\sigma$ is a set of $k+1$ interacting vertices.
        \begin{itemize}
            \item 0-simplex: A single node.
            \item 1-simplex: A pair of nodes (standard edge).
            \item 2-simplex: A group of three nodes (triangle/hyperedge).
        \end{itemize}
        \item \textbf{Epidemic Model:} An extension of the Susceptible-Infected-Susceptible (SIS) model.
        \item \textbf{Simplicial Contagion:} Contagion rates depend on interaction dimensionality. 
        \begin{itemize}
            \item Pairwise infection rate: $\beta$
            \item 2-simplex (group) infection rate: $\beta_{\Delta}$
        \end{itemize}
    \end{itemize}

\begin{frame}{Experimental Setup}
    \begin{itemize}
        \item \textbf{Graph Structure:} Simulations run on Erdős-Rényi-style Random Simplicial Complexes (RSCs).
        \item \textbf{Rescaled Infectivities:}
        $$ \lambda = \frac{\beta \langle k \rangle}{\mu} \quad \text{and} \quad \lambda_{\Delta} = \frac{\beta_{\Delta} \langle k_{\Delta} \rangle}{\mu} $$
        \item \textbf{Mean-Field Approach (Homogeneous Mixing):}
        \item Theoretical Final Node Density ($\rho^*$):
        $$ \rho^* = \frac{(\lambda_{\Delta} - \lambda) + \sqrt{(\lambda - \lambda_{\Delta})^2 - 4\lambda_{\Delta}(1-\lambda)}}{2\lambda_{\Delta}} $$
        \item Initial Node Density Threshold ($\rho_c$):
        $$ \rho_c = \frac{(\lambda_{\Delta} - \lambda) - \sqrt{(\lambda - \lambda_{\Delta})^2 - 4\lambda_{\Delta}(1-\lambda)}}{2\lambda_{\Delta}} $$
    \end{itemize}
\end{frame}

\begin{frame}{Precision Error from Homogeneous Mixing Assumption}
    \begin{columns}
        \column{0.4\textwidth}
        \centering
        \includegraphics[width=\linewidth]{fig3.png} \\
        {\footnotesize Noise Reduction ($N=8000$)}
        
        \column{0.6\textwidth}
        \begin{itemize}
            \item \textbf{Testing the Anomaly:} Increased RSC size from $N=2000$ to $N=8000$.
            \item \textbf{Ruling out noise:} If the $N=2000$ anomaly was finite-size stochastic noise, increasing the scale should force the simulations to adhere perfectly to the theoretical threshold.
            \item \textbf{Conclusion:} They still converged to the endemic state below the theoretical line. 
            \item The theoretical threshold overestimates the true threshold because the \textit{homogeneous mixing assumption} breaks down. Local structural fluctuations in the quenched network act as denser ignition points for complex contagion.
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Future Work}
    \begin{itemize}
        \item \textbf{Realistic Network Topologies:} Test the Simplicial Contagion Model on networks exhibiting standard social network properties.
        \begin{itemize}
            \item Power-law/Exponential degree distributions.
            \item Dense community structures.
        \end{itemize}
        \item \textbf{Algorithmic Application:} Optimize the deliberate seeding of infected nodes.
        \begin{itemize}
            \item \textit{Hypothesis:} Strategic random seeding within targeted communities biased by node degree will significantly lower the initial infection density threshold required to trigger a cascade.
        \end{itemize}
    \end{itemize}
