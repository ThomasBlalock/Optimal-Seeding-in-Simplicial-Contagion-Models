\section{Introduction to CIC3}
\begin{frame}
\frametitle{Motivation: Capacity-Constrained Cascades}
\begin{center}
\includegraphics[width=0.9\textwidth]{figures/capacity_constraints.png}
\end{center}
\begin{itemize}
    \item Many social processes have hard quotas (e.g., volleyball needs exactly 12 players)
    \item Value plateaus after quota is met; additional participants may even detract
    \item \textbf{C3}: A single capacity-constrained contagion
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{CIC3: Multiple Competing Cascades}
\begin{center}
\includegraphics[width=0.9\textwidth]{figures/cic3_competition.png}
\end{center}
\begin{itemize}
    \item Multiple C3s compete simultaneously for the same pool of attendees
    \item Each contagion has its own quota $Q_i$
    \item Example: Multiple social events happening at the same time
    \item \textbf{Objective}: Maximize global attainment across all contagions
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Global Attainment Definitions}
\begin{block}{Per-Contagion Attainment}
Let $I_i^{raw}$ be nodes infected by contagion $C_i$, $Q_i$ its quota.
$$K_i = \min(|I_i^{raw}|, Q_i)$$
$$A_i = \frac{K_i}{Q_i} \in [0,1]$$
\end{block}

\begin{block}{Global Attainment}
$$A_g = \frac{1}{|\mathcal{C}|} \sum_{C_i \in \mathcal{C}} A_i \in [0,1]$$
\end{block}

\begin{block}{Time-Discounted Attainment}
With value function $V(t) \in [0,1]$:
$$K_i^{td} = \min\left(\sum_{k=1}^{\min(Q_i,|I_i^{raw}|)} V(t_{i,(k)}), Q_i\right)$$
$$A_i^{td} = \frac{K_i^{td}}{Q_i}, \quad A_g^{td} = \frac{1}{|\mathcal{C}|} \sum_{C_i \in \mathcal{C}} A_i^{td}$$
\end{block}
\end{frame}

\section{Experimental Setup}
\begin{frame}
\frametitle{Network Topologies}
\begin{center}

    \item \textbf{BA}: Barabási–Albert with simplicial extension
    \item \textbf{SBM}: Stochastic block model with community structure
    \item \textbf{SBM-Manual}: Tuned to stress clustering strategies (53 communities)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{SBM-Manual Configuration}
\begin{center}
\includegraphics[width=0.9\textwidth]{figures/sbm_manual_config.png}
\end{center}
\begin{itemize}
    \item $N = 500$, $K = 53$ communities
    \item 52 small cliques ($p_{intra}=1.0$, $p_{inter}=0.003$)
    \item 1 hub community (size 32, $p_{intra}=0.5$, $p_{inter}=0.15$)
    \item Triangle probabilities: $[0.2]^{52} + [0.04]$
    \item Realized: $\langle k \rangle \approx 19.05$, $\langle k_\Delta \rangle \approx 6.54$
\end{itemize}
\end{frame}

\section{Results}
\begin{frame}
\frametitle{CIC3 Strategy Comparison}
\begin{center}
\includegraphics[width=0.85\textwidth]{figures/cic3_strategy_comparison.png}
\end{center}
\begin{itemize}
    \item High-degree strategy performs best overall
    \item Clustering strategies underperform on most topologies
    \item Random surprisingly effective on sparse networks
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Per-Contagion Distribution}
\begin{center}
\includegraphics[width=0.85\textwidth]{figures/cic3_per_contagion_distribution.png}
\end{center}
\begin{itemize}
    \item Boxplots show distribution of individual contagion attainments
    \item High-degree strategy achieves higher median and less variance
    \item Clustering strategies show more outliers and lower medians
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Probability Density Functions}
\begin{center}
\includegraphics[width=0.85\textwidth]{figures/cic3_per_contagion_pdf.png}
\end{center}
\begin{itemize}
    \item PDFs reveal shape features boxplots miss

\end{frame}

\section{Key Findings}
\begin{frame}
\frametitle{The Sparsity Anomaly}
\begin{center}
\includegraphics[width=0.85\textwidth]{figures/cic3_topology_statistics.png}
\end{center}
\begin{itemize}
    \item Twitter mutual network is much sparser than other topologies
    \item Random seeding performs best on this sparse network
    \item High-degree seeding underperforms in sparse, community-structured networks
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Sparsity Experiment}
\begin{center}
\includegraphics[width=0.85\textwidth]{placeholder_sparsity.png}
\end{center}
\begin{itemize}
    \item Varying inter-community and intra-community connection probabilities
    \item Random works best only in high intra, low inter-community networks
    \item High-degree optimal for dense networks with community structure
\end{itemize}
\end{frame}

\section{Conclusions}
\begin{frame}
\frametitle{Conclusions}
\begin{center}
\includegraphics[width=0.85\textwidth]{figures/conclusions.png}
\end{center}
\begin{itemize}
    \item Centrality-based seeding (degree, triangle) generally superior
    \item Exception: sparse networks with strong community structure
    \item Random seeding can outperform centrality-based methods in specific regimes
    \item Community-based strategies need refinement for sparse regimes
\end{itemize}
\end{frame}
